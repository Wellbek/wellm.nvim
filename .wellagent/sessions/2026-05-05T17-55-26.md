### 2. `lua/wellm/usage.lua` — Use alias resolution + fractional display

Replace the `M.cost_for` function and add a resolver helper:

```lua
-- Resolve a model name to a pricing key (exact match first, then aliases).
local function resolve_model(model)
  local pricing = require("wellm.config").pricing
  if pricing[model] then return model end
  local aliases = require("wellm.config").pricing_aliases
  return aliases[model] or model
end

function M.cost_for(model, input_tok, output_tok)
  local pricing = require("wellm.config").pricing
  local key     = resolve_model(model)
  local p       = pricing[key]
  if not p then return 0 end
  return (input_tok / 1e6) * p.input + (output_tok / 1e6) * p.output
end
```

Also update the `M.summary` function so the displayed model name shows the resolved pricing key alongside the raw model name, so users can see if an alias was used:

```lua
function M.summary(month)
  month    = month or os.date("%Y-%m")
  local d  = load()
  local md = d[month] or {}

  local models      = {}
  local total_cost  = 0
  local total_in    = 0
  local total_out   = 0

  for model, tok in pairs(md) do
    local resolved = resolve_model(model)
    local cost = M.cost_for(model, tok.input, tok.output)
    local display_name = (resolved ~= model)
        and (model .. " → " .. resolved)
        or model
    table.insert(models, {
      model      = display_name,
      input_tok  = tok.input,
      output_tok = tok.output,
      cost_usd   = cost,
    })
    total_cost = total_cost + cost
    total_in   = total_in   + tok.input
    total_out  = total_out  + tok.output
  end

  table.sort(models, function(a, b) return a.cost_usd > b.cost_usd end)

  return {
    month      = month,
    models     = models,
    total_cost = total_cost,
    total_in   = total_in,
    total_out  = total_out,
  }
end
```

### 3. `lua/wellm/ui/usage.lua` — Better formatting for small amounts

The current `%.4f` format shows `$0.0000` for anything under $0.00005 (e.g., 500 tokens on a $3/MTok model = $0.0015, which *does* show — but barely). Use a smarter formatter and add a per-1K breakdown so low-usage is still meaningful:

```lua
-- Replace the local function fmt_tok at the top with these two helpers:
local function fmt_tok(n)
  if n >= 1e6 then return string.format("%.1fM", n / 1e6) end
  if n >= 1e3 then return string.format("%.1fk", n / 1e3) end
  return tostring(n)
end

local function fmt_cost(usd)
  if usd == 0 then return "$0.00" end
  if usd < 0.01 then return string.format("¢%.2f", usd * 100) end   -- e.g. ¢0.15
  if usd < 1    then return string.format("$%.4f", usd) end          -- e.g. $0.1234
  return string.format("$%.2f", usd)                                  -- e.g. $12.34
end
```
